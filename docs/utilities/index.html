Space

.u-{property}{side}-{size}

property:

  • m: set margin
  • .p: set padding

side:

  • t: set margin-top or padding-top
  • r: set margin-right or padding-right
  • b: set margin-bottom or padding-bottom
  • l: set margin-left or padding-left
  • tb: set margin-top and margin-bottom or padding-top and padding-bottom
  • lr: set margin-left and margin-right or padding-left and padding-right

size:

  • 0: set margin or padding to 0
  • 5: set margin or padding to 5px
  • 10: set margin or padding to 10px
  • ...
  • 60: set margin or padding to 60px
{% highlight css %} .u-m0 { margin: 0 !important; } .u-pr5 { padding-right: 5px !important; } .u-mlr15 { margin-left: 15px !important; margin-right: 15px !important; } {% endhighlight %}
